Information about an AI model available on OpenRouter

## Example Usage

```typescript expandable lines
import { Model } from "@openrouter/sdk/models";

let value: Model = {
  id: "openai/gpt-4",
  canonicalSlug: "openai/gpt-4",
  name: "GPT-4",
  created: 1692901234,
  pricing: {
    prompt: "0.00003",
    completion: "0.00006",
  },
  contextLength: 8192,
  architecture: {
    modality: "text->text",
    inputModalities: [
      "text",
    ],
    outputModalities: [
      "text",
    ],
  },
  topProvider: {
    isModerated: true,
  },
  perRequestLimits: null,
  supportedParameters: [
    "temperature",
    "top_p",
    "max_tokens",
  ],
  defaultParameters: null,
};
```
