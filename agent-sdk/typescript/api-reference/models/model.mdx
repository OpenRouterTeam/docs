Information about an AI model available on OpenRouter

## Example Usage

```typescript expandable lines
import { Model } from "@openrouter/sdk/models";

let value: Model = {
  architecture: {
    inputModalities: [
      "text",
    ],
    modality: "text->text",
    outputModalities: [
      "text",
    ],
  },
  canonicalSlug: "openai/gpt-4",
  contextLength: 8192,
  created: 1692901234,
  defaultParameters: null,
  id: "openai/gpt-4",
  links: {
    details: "/api/v1/models/openai/gpt-5.4/endpoints",
  },
  name: "GPT-4",
  perRequestLimits: null,
  pricing: {
    completion: "0.00006",
    prompt: "0.00003",
  },
  supportedParameters: [
    "temperature",
    "top_p",
    "max_tokens",
  ],
  supportedVoices: null,
  topProvider: {
    isModerated: true,
  },
};
```
