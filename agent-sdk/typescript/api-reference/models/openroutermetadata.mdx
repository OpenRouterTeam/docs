## Example Usage

```typescript lines
import { OpenRouterMetadata } from "@openrouter/sdk/models";

let value: OpenRouterMetadata = {
  attempt: 1,
  endpoints: {
    available: [
      {
        model: "openai/gpt-4o",
        provider: "OpenAI",
        selected: true,
      },
    ],
    total: 1,
  },
  isByok: false,
  region: "iad",
  requested: "openai/gpt-4o",
  strategy: "direct",
  summary: "available=1, selected=OpenAI",
};
```
